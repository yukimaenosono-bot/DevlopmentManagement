# 04. ER図

## エンティティ関連図

```mermaid
erDiagram
    %% マスタ
    品目マスタ {
        varchar 品目コード PK
        varchar 品目名
        varchar 品目名略称
        code 品目区分
        code ロット管理区分
        code シリアル管理区分
        code 単位
        decimal 標準単価
        boolean 有効期限管理
        decimal 安全在庫数
        boolean 有効フラグ
    }

    取引先マスタ {
        varchar 取引先コード PK
        varchar 取引先名
        code 取引先区分
        varchar 住所
        varchar 電話番号
        boolean 有効フラグ
    }

    工程マスタ {
        varchar 工程コード PK
        varchar 工程名
        code 工程区分
        boolean 有効フラグ
    }

    設備マスタ {
        varchar 設備コード PK
        varchar 設備名
        varchar 工程コード FK
        boolean 有効フラグ
    }

    倉庫マスタ {
        varchar 倉庫コード PK
        varchar 倉庫名
        code 倉庫区分
    }

    棚マスタ {
        varchar 棚コード PK
        varchar 倉庫コード FK
        varchar 棚名
    }

    BOM {
        varchar 親品目コード PK
        varchar 子品目コード PK
        decimal 必要数量
        code 単位
        date 有効開始日
        date 有効終了日
    }

    工程ルーティング {
        varchar 品目コード PK
        int 工程順序 PK
        varchar 工程コード FK
        decimal 標準時間
        boolean 有効フラグ
    }

    検査基準マスタ {
        varchar 検査基準ID PK
        varchar 品目コード FK
        varchar 工程コード FK
        varchar 検査項目名
        varchar 合格基準
        varchar 単位
    }

    %% トランザクション
    受注 {
        varchar 受注番号 PK
        varchar 取引先コード FK
        date 受注日
        date 納期
        varchar ステータス
    }

    受注明細 {
        varchar 受注番号 PK
        int 明細番号 PK
        varchar 品目コード FK
        decimal 数量
        decimal 単価
    }

    生産計画 {
        varchar 計画番号 PK
        varchar 受注番号 FK
        varchar 品目コード FK
        decimal 計画数量
        date 計画開始日
        date 計画完了日
        varchar ステータス
    }

    製造指示 {
        varchar 製造指示番号 PK
        varchar 計画番号 FK
        varchar 品目コード FK
        varchar 製造ロット番号
        decimal 指示数量
        date 製造開始予定日
        date 完了予定日
        varchar ステータス
        varchar 作業指示コメント
        timestamp 作成日時
        varchar 作成者
    }

    工程実績 {
        varchar 実績ID PK
        varchar 製造指示番号 FK
        varchar 工程コード FK
        varchar 設備コード FK
        int 工程順序
        timestamp 着手日時
        timestamp 完了日時
        decimal 完了数量
        decimal 不良数量
        varchar 担当者
        varchar ステータス
    }

    在庫 {
        varchar 在庫ID PK
        varchar 品目コード FK
        varchar 倉庫コード FK
        varchar 棚コード FK
        varchar ロット番号
        decimal 在庫数量
        code 在庫区分
        date 有効期限
        date 入庫日
    }

    入出庫履歴 {
        varchar 履歴ID PK
        varchar 品目コード FK
        varchar 倉庫コード FK
        varchar ロット番号
        code 入出庫区分
        decimal 数量
        varchar 関連伝票番号
        timestamp 処理日時
        varchar 担当者
    }

    検査実績 {
        varchar 検査ID PK
        varchar 製造指示番号 FK
        varchar 工程コード FK
        date 検査日
        decimal 検査数量
        decimal 不良数量
        code 総合合否
        varchar 担当者
        varchar コメント
    }

    検査明細 {
        varchar 検査ID PK
        varchar 検査基準ID PK
        decimal 測定値
        code 合否
    }

    不良情報 {
        varchar 不良ID PK
        varchar 検査ID FK
        varchar 製造指示番号 FK
        varchar 不良コード
        decimal 不良数量
        code 処置区分
        boolean 特採承認フラグ
        varchar 承認者
        varchar 原因
        varchar 対策
    }

    出荷指示 {
        varchar 出荷指示番号 PK
        varchar 受注番号 FK
        varchar 取引先コード FK
        date 出荷予定日
        varchar ステータス
    }

    出荷指示明細 {
        varchar 出荷指示番号 PK
        int 明細番号 PK
        varchar 品目コード FK
        decimal 指示数量
    }

    出荷実績 {
        varchar 出荷実績番号 PK
        varchar 出荷指示番号 FK
        date 出荷日
        varchar 運送会社
        varchar 送り状番号
        varchar 担当者
    }

    出荷実績明細 {
        varchar 出荷実績番号 PK
        int 明細番号 PK
        varchar 品目コード FK
        varchar ロット番号 FK
        decimal 出荷数量
    }

    %% リレーション
    品目マスタ ||--o{ BOM : "親品目"
    品目マスタ ||--o{ BOM : "子品目"
    品目マスタ ||--o{ 工程ルーティング : ""
    品目マスタ ||--o{ 検査基準マスタ : ""
    工程マスタ ||--o{ 設備マスタ : ""
    工程マスタ ||--o{ 工程ルーティング : ""
    工程マスタ ||--o{ 検査基準マスタ : ""
    倉庫マスタ ||--o{ 棚マスタ : ""

    取引先マスタ ||--o{ 受注 : ""
    受注 ||--o{ 受注明細 : ""
    受注明細 }o--|| 品目マスタ : ""
    受注 ||--o{ 生産計画 : ""
    生産計画 }o--|| 品目マスタ : ""
    生産計画 ||--o{ 製造指示 : ""
    製造指示 }o--|| 品目マスタ : ""
    製造指示 ||--o{ 工程実績 : ""
    工程実績 }o--|| 工程マスタ : ""
    工程実績 }o--o| 設備マスタ : ""

    品目マスタ ||--o{ 在庫 : ""
    倉庫マスタ ||--o{ 在庫 : ""
    棚マスタ ||--o{ 在庫 : ""
    品目マスタ ||--o{ 入出庫履歴 : ""

    製造指示 ||--o{ 検査実績 : ""
    工程マスタ ||--o{ 検査実績 : ""
    検査実績 ||--o{ 検査明細 : ""
    検査基準マスタ ||--o{ 検査明細 : ""
    検査実績 ||--o{ 不良情報 : ""

    取引先マスタ ||--o{ 出荷指示 : ""
    受注 ||--o{ 出荷指示 : ""
    出荷指示 ||--o{ 出荷指示明細 : ""
    出荷指示明細 }o--|| 品目マスタ : ""
    出荷指示 ||--o| 出荷実績 : ""
    出荷実績 ||--o{ 出荷実績明細 : ""
    出荷実績明細 }o--|| 品目マスタ : ""
```
