# 01. 画面遷移図

## 全体画面遷移

```mermaid
flowchart TD
    LOGIN([ログイン画面]) --> DASHBOARD[ダッシュボード]

    DASHBOARD --> MO[製造指示管理]
    DASHBOARD --> PP[生産計画管理]
    DASHBOARD --> INV[在庫管理]
    DASHBOARD --> QC[品質管理]
    DASHBOARD --> PROC[工程管理]
    DASHBOARD --> SH[出荷管理]
    DASHBOARD --> MST[マスタ管理]
    DASHBOARD --> SYS[システム管理]

    subgraph MO[製造指示管理]
        MO_LIST[製造指示一覧] --> MO_NEW[製造指示発行]
        MO_LIST --> MO_DETAIL[製造指示詳細・編集]
        MO_LIST --> MO_FIELD[現場端末用表示]
        MO_NEW --> MO_LIST
        MO_DETAIL --> MO_LIST
    end

    subgraph PP[生産計画管理]
        PP_LIST[生産計画一覧] --> PP_NEW[生産計画作成]
        PP_LIST --> PP_DETAIL[生産計画詳細・編集]
        PP_NEW --> PP_LIST
        PP_DETAIL --> MO_NEW
    end

    subgraph INV[在庫管理]
        INV_LIST[在庫一覧・照会] --> INV_IN[入庫登録]
        INV_LIST --> INV_OUT[出庫登録]
        INV_LIST --> INV_COUNT[棚卸]
        INV_IN --> INV_LIST
        INV_OUT --> INV_LIST
    end

    subgraph QC[品質管理]
        QC_LIST[検査実績一覧] --> QC_INPUT[検査結果入力]
        QC_LIST --> QC_DEFECT[不良登録・処置]
        QC_LIST --> QC_ANALYSIS[品質分析]
        QC_INPUT --> QC_LIST
        QC_DEFECT --> QC_LIST
    end

    subgraph PROC[工程管理]
        PROC_MONITOR[工程進捗モニタリング] --> PROC_INPUT[工程実績入力]
        PROC_INPUT --> PROC_MONITOR
    end

    subgraph SH[出荷管理]
        SH_LIST[出荷指示一覧] --> SH_DETAIL[出荷指示詳細]
        SH_DETAIL --> SH_RESULT[出荷実績登録]
        SH_DETAIL --> SH_DELIVERY[納品書発行]
        SH_RESULT --> SH_LIST
    end

    subgraph MST[マスタ管理]
        MST_ITEM[品目マスタ]
        MST_PARTNER[取引先マスタ]
        MST_PROCESS[工程マスタ]
        MST_BOM[BOM（部品表）]
        MST_ROUTING[工程ルーティング]
        MST_EQUIP[設備マスタ]
        MST_WAREHOUSE[倉庫・棚マスタ]
        MST_INSPECT[検査基準マスタ]
    end

    subgraph SYS[システム管理]
        SYS_USER[ユーザー管理]
        SYS_ROLE[権限・ロール管理]
        SYS_LOG[操作ログ照会]
    end
```
